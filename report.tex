% ── 1.1  Entities ─────────────────────────────────────────────
\subsection{Entities}

\begin{itemize}
  \item \textbf{ORGANIZATION} -- A US federal agency / publishing body on
        data.gov.  PK: \texttt{name} (slug).
  \item \textbf{CONTACT} -- A contact person assigned to an organization
        who may maintain one or more datasets.  PK: \texttt{email}.
  \item \textbf{DATASET} -- A dataset (package) published on data.gov.
        PK: \texttt{name} (slug).
  \item \textbf{RESOURCE} -- A downloadable file or API endpoint identified
        globally by its URL.  PK: \texttt{url}.
  \item \textbf{APP\_USER} -- A registered application user.
        PK: \texttt{email}.\enspace \texttt{age} is a
        \emph{derived attribute} computed from \texttt{birthdate}.
  \item \textbf{TOPIC} -- A topic / theme that datasets can belong to.
        PK: \texttt{name} (e.g.\ ``Transportation'', ``Health'').
  \item \textbf{PROJECT} -- A weak entity representing a user's project.
        Identified through APP\_USER via the \emph{creates} identifying
        relationship.  Partial key: \texttt{name}.
\end{itemize}

\section{Relational Model}
% =============================================================

All primary keys are \textbf{natural} -- no synthetic auto-increment IDs.
\underline{Underlined} attributes form the primary key;
\textit{italicised} attributes denote foreign keys.

\bigskip
\noindent\textbf{URL collision convention for RESOURCE metadata}
(same URL appearing across multiple datasets):
keep one \textsc{resource} row per \texttt{url};
first non-null value wins for text metadata;
earliest non-null for \texttt{created} datetime.

\begin{enumerate}
  \item \textbf{ORGANIZATION}(\underline{name}, title, description,
        org\_type, image\_url, created)

  \item \textbf{CONTACT}(\underline{email}, name, \textit{org\_name})\\
        org\_name $\rightarrow$ ORGANIZATION.name

  \item \textbf{DATASET}(\underline{name}, title, description,
        access\_level, license\_id, license\_title, metadata\_created,
        metadata\_modified, identifier, publisher\_name, landing\_page,
        modified, issued, accrual\_periodicity, temporal,
        bureau\_code, program\_code,
        \textit{org\_name}, \textit{maintainer\_email})\\
        org\_name $\rightarrow$ ORGANIZATION.name\\
        maintainer\_email $\rightarrow$ CONTACT.email

  \item \textbf{RESOURCE}(\underline{url}, name, description, format,
        mimetype, created)

  \item \textbf{DATASET\_RESOURCE}(%
        \underline{\textit{dataset\_name}},
        \underline{\textit{url}}) -- M:N bridge\\
        dataset\_name $\rightarrow$ DATASET.name\\
        url $\rightarrow$ RESOURCE.url

  \item \textbf{DATASET\_TAG}(%
        \underline{\textit{dataset\_name}},
        \underline{tag\_name}) -- multivalued attribute\\
        dataset\_name $\rightarrow$ DATASET.name

  \item \textbf{TOPIC}(\underline{name}, description)

  \item \textbf{DATASET\_TOPIC}(%
        \underline{\textit{dataset\_name}},
        \underline{\textit{topic\_name}}) -- M:N bridge\\
        dataset\_name $\rightarrow$ DATASET.name\\
        topic\_name $\rightarrow$ TOPIC.name

  \item \textbf{APP\_USER}(\underline{email}, username, gender,
        birthdate, country)\\
        age = derived from birthdate

  \item \textbf{PROJECT}(%
        \underline{\textit{email}},
        \underline{name},
        type) -- weak entity\\
        email $\rightarrow$ APP\_USER.email

  \item \textbf{PROJECT\_DATASET}(%
        \underline{\textit{email}},
        \underline{\textit{project\_name}},
        \underline{\textit{dataset\_name}}) -- M:N ``uses''\\
        (email, project\_name) $\rightarrow$ PROJECT.(email, name)\\
        dataset\_name $\rightarrow$ DATASET.name
\end{enumerate}
